# smartgrader
Linux:
# Creează venv
python3 -m venv venv

# Activează venv
source venv/bin/activate

# Instalare Dependințe
pip install -r requirements.txt

Windows:

# Creează venv
python -m venv venv

# Activează venv (Command Prompt)
venv\Scripts\activate.bat

# SAU activează venv (PowerShell)
venv\Scripts\Activate.ps1

# Instalare Dependințe
pip install -r requirements.txt
